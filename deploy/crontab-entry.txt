# Optional cron entry for the daily pipeline run on the Droplet.
#
# The FastAPI app already runs APScheduler in-process (see app/config.py —
# DAILY_RUN_HOUR / DAILY_RUN_MINUTE, both UTC). If uvicorn is always running,
# APScheduler alone is sufficient and no cron entry is needed.
#
# Add this cron line only if you want an independent OS-level safety net
# (e.g. in case uvicorn crashes before the daily trigger fires):
#
#   crontab -e        # as the uvps user
#   # then paste the line below:
#
# Runs at 03:10 UTC (10 minutes after APScheduler fires, acts as a fallback).
# Redirect stdout/stderr to a rotating log so the file doesn't grow unbounded.

10 3 * * * cd /home/uvps/uvps && /home/uvps/uvps/.venv/bin/python -m app.run_now >> /var/log/uvps-cron.log 2>&1

# To enable log rotation for that file, create /etc/logrotate.d/uvps-cron:
#
#   /var/log/uvps-cron.log {
#       daily
#       rotate 7
#       compress
#       missingok
#       notifempty
#   }
